Wire detect_closure_prophecies into vcgen.rs so FnMut closures with ByMutRef captures emit
prophecy variable SMT declarations in every generated VC script.

Plan 01; here we integrate them into the declaration collection phase of generate_vcs so that
contracts like `#[ensures(count == old(count) + 1)]` on FnMut closures can reference
`count_initial` and `count_prophecy` SMT variables.

   // Wire prophecy declarations for FnMut closures with mutable captures
   let mut closure_prophecy_decls: Vec<rust_fv_smtlib::command::Command> = Vec::new();
   for ci in &closure_infos {
       let cp = crate::encode_prophecy::detect_closure_prophecies(ci);

           let mut decls = crate::encode_prophecy::prophecy_declarations(&cp);
           closure_prophecy_decls.append(&mut decls);
