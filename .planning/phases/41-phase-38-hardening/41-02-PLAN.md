<objective>
Wire dynamic dispatch call-site VCs to trait-level contracts by normalizing callee names of the form `<dyn TraitName>::method` in `generate_call_site_vcs` to look up the matching contract in `contract_db` by trait method name, instead of falling through to the OpaqueCallee diagnostic path.

Purpose: TRT-02 requires that `dyn Trait` call sites use trait-level contracts for verification. Currently, MIR terminators for `dyn Trait` calls produce callee names like `<dyn Stack>::push` which do not match any `contract_db` key, triggering OpaqueCallee. The fix normalizes these names before the `contract_db.get()` lookup.

Output: `vcgen.rs` with dyn-dispatch callee name resolution + integration test in `trait_verification.rs`.

/// If `callee_name` is a dyn-dispatch call (e.g. "<dyn Stack>::push"),
/// extract (trait_name, method_name) = ("Stack", "push").
fn parse_dyn_dispatch_callee(callee_name: &str) -> Option<(&str, &str)> {
    // Match "<dyn TraitName>::method" or "<dyn TraitName as TraitName>::method"
    let inner = callee_name.strip_prefix("<dyn ")?;
    // split on ">:: " to find method
    let gt_pos = inner.find('>')?;
    let trait_part = &inner[..gt_pos];
    // trait_part may be "Stack" or "Stack as Stack"
    let trait_name = trait_part.split(" as ").next()?.trim();
    let method_name = callee_name.rsplit("::").next()?;
    Some((trait_name, method_name))
}
```

/// Extract (trait_name, method_name) from a dyn-dispatch callee name.
///
/// Handles patterns produced by rustc MIR for virtual dispatch:
/// - `<dyn Stack>::push` → ("Stack", "push")
/// - `<dyn Stack as Stack>::push` → ("Stack", "push")
///
/// Returns None for non-dyn callee names.
fn parse_dyn_dispatch_callee(callee_name: &str) -> Option<(&str, &str)> {
    let inner = callee_name.strip_prefix("<dyn ")?;
    let gt_pos = inner.find('>')?;
    let trait_part = &inner[..gt_pos];
    // Handle "<dyn Stack as Stack>" — take the part before " as "
    let trait_name = trait_part.split(" as ").next()?.trim();
    let method_name = callee_name.rsplit("::").next()?;
    if method_name.is_empty() {
        return None;
    }
    Some((trait_name, method_name))
}
```

///
/// When a function calls a method via `dyn Trait` (callee_name = "<dyn Stack>::push"),
/// generate_call_site_vcs must look up Stack::push in contract_db and emit a contract-based
/// VC — NOT an OpaqueCallee diagnostic.
#[test]
fn dyn_dispatch_call_site_uses_trait_contracts() {
    use rust_fv_analysis::vcgen::{generate_vcs_with_db, VcKind};
    use rust_fv_analysis::ir::{
        BasicBlock, Contracts, Function, IntTy, Local, Operand, Place, Rvalue,
        Statement, Terminator, Ty,
    };
    use rust_fv_analysis::contract_db::{ContractDatabase, FunctionSummary, Contracts as DbContracts, SpecExpr as DbSpecExpr};
    use rust_fv_analysis::ghost_predicates::GhostPredicateDatabase;

    // Build a function that calls <dyn Stack>::push via dynamic dispatch
    let func = Function {
        name: "caller".to_string(),
        params: vec![Local::new("_1", Ty::TraitObject("Stack".to_string()))],
        return_local: Local::new("_0", Ty::Unit),
        locals: vec![],
        basic_blocks: vec![BasicBlock {
            statements: vec![],
            terminator: Terminator::Call {
                func: "<dyn Stack>::push".to_string(),
                args: vec![Operand::Copy(Place::local("_1"))],
                destination: Place::local("_0"),
                target: 1,
            },
        }, BasicBlock {
            statements: vec![],
            terminator: Terminator::Return,
        }],
        contracts: Contracts::default(),
        generic_params: vec![],
        loops: vec![],
        prophecies: vec![],
        lifetime_params: vec![],
        outlives_constraints: vec![],
        borrow_info: vec![],
        reborrow_chains: vec![],
        unsafe_blocks: vec![],
        unsafe_operations: vec![],
        unsafe_contracts: None,
        is_unsafe_fn: false,
        thread_spawns: vec![],
        atomic_ops: vec![],
        sync_ops: vec![],
        lock_invariants: vec![],
        concurrency_config: None,
        source_names: std::collections::HashMap::new(),
        coroutine_info: None,
    };

    // Build contract_db with Stack::push having a requires contract
    let mut contract_db = ContractDatabase::new();
    let push_summary = FunctionSummary {
        contracts: DbContracts {
            requires: vec![DbSpecExpr { raw: "x > 0".to_string() }],
            ensures: vec![],
            ..DbContracts::default()
        },
        param_names: vec!["x".to_string()],
        param_types: vec![Ty::Int(IntTy::I32)],
        return_ty: Ty::Unit,
        ..FunctionSummary::default()
    };
    contract_db.insert("Stack::push".to_string(), push_summary);

    let ghost_db = GhostPredicateDatabase::new();
    let vcs = generate_vcs_with_db(&func, Some(&contract_db), &ghost_db);

    // Should have at least one VC from the trait contract (not OpaqueCallee)
    let non_opaque_vcs: Vec<_> = vcs
        .conditions
        .iter()
        .filter(|vc| !matches!(vc.location.vc_kind, VcKind::OpaqueCallee | VcKind::OpaqueCalleeUnsafe))
        .collect();

    assert!(
        !non_opaque_vcs.is_empty(),
        "Expected at least one contract-based VC from dyn dispatch resolution; got only opaque diagnostics. VCs: {:#?}",
        vcs.conditions.iter().map(|vc| (&vc.description, &vc.location.vc_kind)).collect::<Vec<_>>()
    );
}
