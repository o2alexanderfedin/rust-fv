Rvalue::Repeat(operand, count) => {
    let val = encode_operand_for_vcgen(operand, func);
    let count_val = *count as i128;
    let arr_name = place.local.clone();
    let len_name = crate::encode_term::len_constant_name(&arr_name);

    // Emit length assertion: arr_len = N
    // (This is handled separately — we return the forall assertion here)

    // For known N, encode as: (forall ((i Int)) (=> (and (>= i 0) (< i N)) (= (select arr i) val)))
    let i_var = "repeat_idx".to_string();
    let i_term = Term::Const(i_var.clone());
    let arr_term = Term::Const(arr_name.clone());

    let select = Term::Select(Box::new(arr_term), Box::new(i_term.clone()));
    let body = Term::Eq(Box::new(select), Box::new(val));

    let in_bounds = Term::And(vec![
        Term::IntGe(Box::new(i_term.clone()), Box::new(Term::IntLit(0))),
        Term::IntLt(Box::new(i_term), Box::new(Term::IntLit(count_val))),
    ]);

    let forall_body = Term::Implies(Box::new(in_bounds), Box::new(body));
    let forall = Term::Forall(
        vec![(String::from("repeat_idx"), Sort::Int)],
        Box::new(forall_body),
    );

    // Return the forall as the rhs (will be wrapped in Assert(Eq(lhs, ...)) by caller)
    // Actually, Repeat is different — the array IS the result, so we need to assert the forall directly
    // plus assert len = N. We return early with two assertions or use the pattern below.
    // Since encode_assignment returns a single Command, we'll assert the forall about the array.
    // The lhs = arr pattern doesn't apply to forall — we need to return a direct assert.
    return Some(Command::Assert(forall));
    // Note: Also emit length constant. Add a helper or emit the length assertion
